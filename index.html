function autofill() {
  var sheet = SpreadsheetApp.getActiveSheet();
  var lastRow = sheet.getLastRow()+1;

  for (var i=1; i<=lastRow; i++) {
    var value = sheet.getRange(i,1).getValue();
    var date = sheet.getRange(i,6).getValue();
    var fid = sheet.getRange(i,5).getValue();

    for (var j=sheet.getActiveCell().getRow(); j<=lastRow; j++) {
      if (sheet.getRange(j,1).getValue() == value) {
        sheet.getRange(j,6).setValue(date);
	sheet.getRange(j,5).setValue(fid);
      }
    }
  }
}
